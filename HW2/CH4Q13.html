# merge all above parts into 1 function to test best method and variables combination
def test_combination(method, variables, k=None):

  datatrain = train[variables]
  datatest = test[variables]
  # indices don't change for train and test splits
  truth = ytrain
  gtruth = ytest

  # choose method and get predictions
  match method:
    case "NB":
      NB = GaussianNB()
      NB.fit(datatrain, truth)
      preds = NB.predict(datatest)
    case "LDA":
      lda = LDA(store_covariance=True)
      lda.fit(datatrain, truth)
      preds = lda.predict(datatest)
    case "QDA":
      qda = QDA(store_covariance=True)
      qda.fit(datatrain, truth)
      preds = qda.predict(datatest)
    case "GLM":
      glm = sm.GLM(truth, datatrain, family=sm.families.Binomial())
      results = glm.fit()
      preds = results.predict(datatest)
      preds = np.greater_equal(preds, 0.5).astype(int)
    case "KNN":
      knn1 = KNeighborsClassifier(n_neighbors=k)
      knn1.fit(datatrain, truth)
      preds = knn1.predict(datatest)

  # error rate and confusion matrix
  print('Method: ' + method)
  print(variables)
  if k:
    print(' | K: ' + str(k))
  print('error_rate: {}'.format(np.mean(preds == gtruth)))
  print(sklearn.metrics.confusion_matrix(gtruth, preds))
  print("=================================================")
